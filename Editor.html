<script>
char type;
function bold()
{
	var y = document.getElementById('input');
	var x = y.value;
	y.style.fontWeight='bold';
        type='B';
}
function italics()
{
	var y = document.getElementById('input');
	var x = y.value;
	y.style.fontStyle='italic';
        y.innerHTML = y.value;
        type='i';
}
function underline()
{
        var y = document.getElementById('input');
	var x = y.value;
	y.style.textDecoration='underline';
        y.innerHTML = y.value;
        type='U';
}
function unBold()
{
	var y = document.getElementById('input');
	var x = y.value;
	y.style.fontWeight='normal';
}
function unItalics()
{
	var y = document.getElementById('input');
	var x = y.value;
	y.style.fontStyle='normal';
        y.innerHTML = y.value;
}
function noUnderline()
{
        var y = document.getElementById('input');
	var x = y.value;
	y.style.textDecoration='none';
        y.innerHTML = y.value;
}
function ditalics()
{
        var type='D';
        var y = document.getElementById('input');
	var x = y.value;
        var token = x.split(' ');
        y.style.fontStyle='italic';
        var ditalicize="";
        for(var i=0; i<token.length; i++)
        {
           if(i%2===0)
           ditalicize+=token[i].toUpperCase;
           else
           ditalicize+=token[i].toLowerCase;
        }
        y.value=ditalicize;
	y.innerHTML = y.value;
        
}       

function callFunctions()
{
       if (type=='B')
       bold();
       if (type=='i')
       italics();
       if (type=='U')
       underline();
       if (type=='O')
       dold();
       if(type=='D')
       ditalics();
}
